CREATE TABLE autor (
    id_autor SERIAL PRIMARY KEY,
    nome VARCHAR(200),
    nacionalidade VARCHAR(200),
    dt_nascimento DATE
);

CREATE TABLE livro (
    id_livro SERIAL PRIMARY KEY,
    titulo VARCHAR(200),
    genero VARCHAR(200),
    n_paginas INT,
    id_autor INT,
    CONSTRAINT fk_livro_autor FOREIGN KEY (id_autor) REFERENCES autor(id_autor)
);
